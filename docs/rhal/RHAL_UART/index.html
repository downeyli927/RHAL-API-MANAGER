RHAL_UART api

Custom Data Types​

PORT_STATUS_T​

typedef enum
{
        PORT_STATUS_CLOSE = 0,
        PORT_STATUS_OPEN = 1,
}PORT_STATUS_T;

Function Declaration​

RHAL_UART_GetDftCfg​

DTV_STATUS_T RHAL_UART_GetDftCfg(struct termios*pCfg);

Get default config param for uart

Parameters
pCfgstruct termios*: pointer to the uart param

RHAL_UART_SwitchUart​

DTV_STATUS_T RHAL_UART_SwitchUart(SINT32 port_id);

Switch uart pinmux to another port

Parameters
port_idSINT32: which uart port need switch

RHAL_UART_OpenUart​

DTV_STATUS_T RHAL_UART_OpenUart(SINT32 port_id, struct termios*pCfg);

Open an uart port

Parameters
port_idSINT32: which uart port need open
pCfgstruct termios*: uart param when it open,if pCfg is NULL,will use a default uart param

RHAL_UART_CloseUart​

DTV_STATUS_T RHAL_UART_CloseUart(SINT32 port_id);

Close an uart port

Parameters
port_idSINT32: which uart port need close

RHAL_UART_ReadUart​

DTV_STATUS_T RHAL_UART_ReadUart(SINT32 port_id, UINT8*pbuff, SINT32*content_size, UINT32 read_size);

Read data from a uart port

Parameters
port_idSINT32: which uart port need open
pbuffUINT8*: the buffer to save read content
content_sizeSINT32*: the length read from uart
read_sizeUINT32: max length read from uart

RHAL_UART_WriteUart​

DTV_STATUS_T RHAL_UART_WriteUart(SINT32 port_id, const UINT8*pbuff, UINT32 write_size);

Write data to a uart port

Parameters
port_idSINT32: which uart port need write
pbuffUINT8*: the buffer to save write content
write_sizeUINT32: the content length write to uart

RHAL_UART_GetPortStatus​

DTV_STATUS_T RHAL_UART_GetPortStatus(SINT32 port_id, PORT_STATUS_T*pStat);

Get port open/close status

Parameters
port_idSINT32: which uart port need write
pStatPORT_STATUS_T*: pointer to status data

RHAL_UART_GetPortCfg​

DTV_STATUS_T RHAL_UART_GetPortCfg(SINT32 port_id, struct termios*pCfg);

Get port config param

Parameters
port_idSINT32: which uart port to get
pCfgstruct termios*: pointer to status data

RHAL_UART_SetPortCfg​

DTV_STATUS_T RHAL_UART_SetPortCfg(SINT32 port_id, struct termios*pCfg);

Set port config param

Parameters
port_idSINT32: which uart port to set
pCfgstruct termios*: pointer to status data

DTV_STATUS_T RHAL_UART_GetDftPort(SINT32*port_id);
